zT.clearEmptyItem(Array)  Or  zZ.fn.clearEmptyItem(Array)

清除数组中被"delete"关键字删除的无效空项目,传入要清理的数组，返回清理成功的数组。

说明:

已知有如下javaScript代码：

function clearEmptyArrayItem(){
    var Array  =  ["a","b","c","d","e","f","g","h"]
    for(var i=0;i<Array.length;i++){
        if(Array[i] == 'c'){
            delete Array[i];
        }
    }

}

以上代码是javaScript日常使用中经常碰到的，在循环中使用"delete"关键字删除被筛选到的数组项目。

但是使用"delete"关键字会出现数组元素被删除后留下空项目的问题，如果我们将使用"delete"操作过的数组之项目一一alert出来，我们就会发现一些问题。

请看下列代码：

function clearEmptyArrayItem(){
    var Array  =  ["a","b","c","d","e","f","g","h"]
    alert("这是删除前_length为："+Array.length);
    for(var i=0;i<Array.length;i++){
        alert("这是删除前_当前下标是："+i+" 当前值是:"+Array[i]);
        if(Array[i] == 'c'){
            delete Array[i];
        }
    }
    alert("这是删除后_length为："+Array.length);

    for(var i=0;i<Array.length;i++){
        alert("这是删除后_当前下标是："+i+" 当前值是:"+Array[i]);
    }

}

通过以上代码我们可以观察到删除前和删除后数组值的变化。它们分别是：

删除前:["a","b","c","d","e","f","g","h"]    length:7

删除后:["a","b",undefined,"d","e","f","g","h"]    length:7

可见使用delete 关键字操作数组项目后确实可以“删除”其值，但是却没有将数组项目从其下标中移除，其数组长度也没变。

遇见这种情况只能将其'undefined'的数组项目从数组中剔除出去，如果您认为在您的代码中必须要使用delete关键字，请在delete 关键字操作后，使用zT.clearEmptyItem()清理一下被delete成undefinde的数组项目，代码如下:

function clearEmptyArrayItem(){
    var Array  =  ["a","b","c","d","e","f","g","h"]
    alert("这是删除前_length为："+Array.length);
    for(var i=0;i<Array.length;i++){
        alert("这是删除前_当前下标是："+i+" 当前值是:"+Array[i]);
        if(Array[i] == 'c'){
            delete Array[i];
        }
    }
    alert("这是删除后_length为："+Array.length);

    for(var i=0;i<Array.length;i++){
        alert("这是删除后_当前下标是："+i+" 当前值是:"+Array[i]);
    }
    Array = zT.clearEmptyItem(Array);
    for(var i=0;i<Array.length;i++){
        alert("这是整理后_当前下标是："+i+" 当前值是:"+Array[i]);
    }
    alert("这是整理后_length为："+Array.length);

}

通过zT.clearEmptyItem(Array)将Array数组中的undefined项目彻底删除,我们可以看看整个代码的运行结果对比：

删除前:["a","b","c","d","e","f","g","h"]    length:7

删除后:["a","b",undefined,"d","e","f","g","h"]    length:7

使用zT.clearEmptyItem(Array)整理后:["a","b","d","e","f","g","h"]    length:6
